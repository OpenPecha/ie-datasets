རྒྱུ་ལས་དང་ཉོན་མོངས་པའི་དབང་གིས་འབྲས་བུ་འགྲོ་བ་ལྔར་འཁོར་བས་ནའོ། ། རྒྱ་མཚོ་ཞེས་བྱ་བ་ནི། རྒྱ་མཚོ་དང་འདྲ་བས་ན་རྒྱ་མཚོ་སྟེ། ཇི་ལྟར་རྒྱ་མཚོ་ཆུ་སྲིན་ལ་སོགས་པ་འཇིགས་པ་དུ་མས་གང་ཞིང༌། ཕ་རོལ་གྱི་མཐའ་མི་མཐོང་བ་ཞེས་བྱ་སྟེ། དེ་བཞིན་དུ་རང་གི་རྣམ་པར་རྟོག་པ་འཁོར་བ་འདིར་འདོད་ཆགས་ལ་སོགས་པའི་ཆུ་སྲིན་ལ་སོགས་པ་སྣ་ཚོགས་ཀྱིས་གང་ཞིང༌། ཕ་རོལ་གྱི་མཐའ་མི་མཐོང་བ་དང་འདྲ་བའོ། ། དེ་བས་ན་འཇིགས་ཤིང་བསྒྲལ་བར་དཀའ་བའོ། ། གང་ཞིག་རྣམ་བསྒོམ་པ་ཞེས་བྱ་བ་ནི། འདི་ལས་ཇི་ལྟར་ཐར་བར་འགྱུར། །
ཞེས་བྱ་བའི་བསམ་པའོ། ། དེའི་དོན་ནི་འདི་ཡིན་ཏེ། འདི་ལྟར་ན་བ་དང་གྲང་བ་དང་བཀྲེས་པ་ལ་སོགས་པའི་སྡུག་བསྔལ་ལ་སྡུག་བསྔལ་གྱི་སྡུག་བསྔལ་དུ་ངོ་ཤེས་པ་ནི་འཇིག་རྟེན་པ་ཐམས་ཅད་ལའང་ཡོད་པའི་ཕྱིར། དེ་སྤང་བར་འདོད་པ་ཙམ་ནི་མྱ་ངན་ལས་འདས་པའི་བསམ་པ་མ་ཡིན་ནོ། ། དེ་ལྟ་མོད་ཀྱང་དེ་དག་གི་སྡུག་བསྔལ་ལ་བརྟེན་ཏེ་ངེས་པར་འབྱུང་བ་སྐྱེ་སྟེ། ཇི་སྐད་དུ་སྤྱོད་འཇུག་ལས། སྡུག་བསྔལ་མེད་པར་ངེས་འབྱུང་མེད། ། དེ་བས་སེམས་ཁྱོད་བརྟན་པར་གནོས། ། ཞེས་གསུངས་སོ། ། དེ་ལྟར་སྡུག་བསྔལ་གྱི་སྡུག་བསྔལ་ལ་སྐྱོ་བས་སྡུག་བསྔལ་འདིའི་མཐའ་གང་ལས་བྱུང་ཞེས་ཀུན་དུ་བརྟགས་པ་ན་མི་དགེ་བ་ལས་མངོན་པར་བྱུང་ངོ༌། །

དེའི་ཕྱིར་འདི་ལས་སྒྲུབ་པ་པོ་གང་ཡང་མེད་ཅེས་བྱ་སྟེ་སྔར་བསྟན་པའི་ཚུལ་གྱིས་བསྒོམ་པ་ལ་སོགས་པ་ཡོད་ན་སྒྲུབ་པ་པོ་ཡོད་པར་འགྱུར་བ་ཞིག་ན། དེ་མེད་པའི་ཕྱིར་སྒྲུབ་པ་པོ་སྟེ་རྣལ་འབྱོར་པ་ཞེས་བྱ་བར་རྟོག་པའང་ག་ལ་ཡོད། སྒྲུབ་པ་པོ་མེད་པའི་ཕྱིར་འབྲས་བུའི་རྣམ་པ་ལ་གཏོད་པའང་ག་ལ་ཡོད་དོ། ། དེ་ལྟར་ཐམས་ཅད་ལས་གྲོལ་བས་ན་ཉོན་མོངས་པ་ལ་སོགས་པའི་སྤང་བྱ་རྣམས་ཀྱང་མ་གྲུབ་པའི་ཕྱིར་འདིར་དེ་ལ་དོར་བ་མེད་ཅེས་བྱའོ། ། སྤང་བྱ་མེད་ན་དེ་སྤང་བའི་གཉེན་པོ་བླང་བ་ཞེས་བྱ་བ་འདིའང་ག་ལ་ཡོད་པར་འགྱུར། མདོར་བསྡུ་ན་དམིགས་པ་དང་མཚན་མ་ཐམས་ཅད་ལས་རྣམ་པར་གྲོལ་བ་ཞེས་བྱ་བའི་དོན་ཏོ། ། ཅི་གལ་ཏེ་དེ་ལྟ་བུ་དེ་ནི་སངས་རྒྱས་ཉག་གཅིག་གི་སྤྱོད་ཡུལ་ཡིན་ཏེ། གསང་བ་སངས་རྒྱས་སྤྱོད་ཡུལ་ཏེ། ། དེ་རྣམས་ཀུན་གྱིས་མི་ཤེས་སོ། ། ཞེས་གསུངས་པའི་ཕྱིར་རོ། །

འཁོར་བ་སྤངས་པར་གྱུར་པ་ཡི། ། མྱ་ངན་འདས་ཁྱོད་མི་བཞེད་ཀྱི། ། འཁོར་བ་མི་དམིགས་པ་ཡི་ནི། ། ཞི་ཉིད་མགོན་པོ་ཁྱོད་ཀྱིས་བསྟན། ། ཞེས་གསུངས་པ་ལྟ་བུའོ། ། དེ་ལྟར་ཚིགས་སུ་བཅད་པ་དང་པོས་ནི་ཀུན་རྫོབ་ཀྱི་གནས་ལུགས་འཁོར་བ་བསྟན་ལ། གཉིས་པས་ནི་མྱ་ངན་ལས་འདས་པ་དོན་དམ་པ་བསྟན་ཅིང༌། གསུམ་པས་ནི་བདེན་པ་གཉིས་ཟུང་དུ་ཆུད་པའམ་གཉིས་སུ་མེད་པ་ཞེས་བྱ་བའི་དོན་བསྟན་པས་ན། དངོས་པོ་ལ་དེ་ལྟར་གནས་པ་བཞིན་དུ་བློས་སྒྲོ་འདོགས་ཆོད་པ་ལ་ལྟ་བ་ཞེས་མིང་དུ་གདགས་པར་བྱའོ། ། དེ་ལྟར་སྒྲོ་འདོགས་ཆོད་པར་གྱུར་ཀྱང་མ་བསྒོམས་ན་དེས་ཕན་པར་ག་ལ་འགྱུར།
